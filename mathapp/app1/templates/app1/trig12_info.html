Trigonometric Identities

Reciprocal

\sin{x} = \frac{1}{\csc{x}} \quad \cos{x} = \frac{1}{\sec{x}} \quad \tan{x} = \frac{1}{\cot{x}} \quad \\ \\ \phantom{aaa} \csc{x} = \frac{1}{\sin{x}} \quad \sec{x} = \frac{1}{\cos{x}} \quad \cot{x} = \frac{1}{\tan{x}} \quad

Pythagorean

\sin{^{2}x} + \cos{^{2}x} = 1 \\ \\ \phantom{aaa} 1 + \tan{^{2}x} = \sec{^{2}x} \\ \\ \phantom{aaa} 1 + \cot{^{2}x} = \csc{^{2}x}

Cofunction

\sin{x} = \cos{\left(\frac{\pi}{2}-x\right)} \quad \cos{x} = \sin{\left(\frac{\pi}{2}-x\right)} \\ \\ \phantom{aaa} \csc{x} = \sec{\left(\frac{\pi}{2}-x\right)} \quad \sec{x} = \csc{\left(\frac{\pi}{2}-x\right)} \\ \\ \phantom{aaa} \tan{x} = \cot{\left(\frac{\pi}{2}-x\right)} \quad \cot{x} = \tan{\left(\frac{\pi}{2}-x\right)} \\ \\

Quotient

\tan{x} = \frac{\sin{x}}{\cos{x}} \quad \cot{x} = \frac{\cos{x}}{\sin{x}}

Double Angle

\sin{(2x)} = 2\sin{x}\cos{x} \\ \\ \phantom{aaa} \cos{(2x)} = \cos{^{2}x} - \sin{^{2}x} \\ \\ \phantom{aaa} \cos{(2x)} = 2\cos{^{2}x} - 1 \\ \\ \phantom{aaa} \cos{(2x)} = 1 - 2\sin{^{2}x} \\ \\ \phantom{aaa}

Sum/Difference

\sin{(a+b)} = \sin{a}\cos{b} + \cos{a}\sin{b} \\ \\ \phantom{aaa} \sin{(a-b)} = \sin{a}\cos{b} - \cos{a}\sin{b} \\ \\ \phantom{aaa} \cos{(a+b)} = \cos{a}\cos{b} - \sin{a}\sin{b} \\ \\ \phantom{aaa} \cos{(a-b)} = \cos{a}\cos{b} + \sin{a}\sin{b} \\ \\ \phantom{aaa} \tan{(a+b)} = \frac{ \tan{a} + \tan{b} }{1 - \tan{a}\tan{b} } \\ \\ \phantom{aaa} \tan{(a-b)} = \frac{ \tan{a} - \tan{b} }{1 + \tan{a}\tan{b} } \\ \\

Product

\sin{a}\cos{b} = \frac{1}{2}[ \sin{(a+b)} + \sin{(a-b)}] \\ \\ \phantom{aaa} \cos{a}\cos{b} = \frac{1}{2}[ \cos{(a-b)} + \cos{(a+b)}]\\ \\ \phantom{aaa} \sin{a}\sin{b} = \frac{1}{2}[ \cos{(a-b)} - \cos{(a+b)}]

cos(x)

f(x) = \cos{x} \\ \\ \phantom{aaa} \mathcal{D}: \{x\epsilon R\} \\ \\ \phantom{aaa} \mathcal{R}: \{y\epsilon R | -1 \leq y \leq 1\} \\ \\ \phantom{aaa} x-ints: \ x = \frac{(2n+1)\pi}{2}, n\epsilon Z \\ \\ \phantom{aaa} y-int: \ y = 1

tan(x)

f(x) = \tan{x} \\ \\ \phantom{aaa} \mathcal{D}: \{x\epsilon R | x \neq \frac{(2n+1)\pi}{2}, n\epsilon Z \} \\ \\ \phantom{aaa} \mathcal{R}: \{y\epsilon R\} \\ \\ \phantom{aaa} x-ints: \ x = n\pi, n\epsilon Z \\ \\ \phantom{aaa} y-int: \ y = 0 \\ \\ \phantom{aaa} HA: x = \frac{(2n+1)\pi}{2}, n\epsilon Z

csc(x)

f(x) = \csc{x} \\ \\ \phantom{aaa} \mathcal{D}: \{x\epsilon R | x \neq n\pi, n\epsilon Z \} \\ \\ \phantom{aaa} \mathcal{R}: \{y\epsilon R | y\leq -1,\ y\geq 1\} \\ \\ \phantom{aaa} x-ints: \ DNE \\ \\ \phantom{aaa} y-int: \ DNE \\ \\ \phantom{aaa} HA: x = n\pi, n\epsilon Z

sec(x)

f(x) = \sec{x} \\ \\ \phantom{aaa} \mathcal{D}: \{x\epsilon R | x \neq \frac{(2n+1)\pi}{2}, n\epsilon Z \} \\ \\ \phantom{aaa} \mathcal{R}: \{y\epsilon R | y\leq -1,\ y\geq 1\} \\ \\ \phantom{aaa} x-ints: \ DNE \\ \\ \phantom{aaa} y-int: y = 1 \\ \\ \phantom{aaa} HA: x = \frac{(2n+1)\pi}{2}, n\epsilon Z

cot(x)
